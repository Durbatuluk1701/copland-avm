Inductive store_event: Ev → Loc → Prop :=
| put_event: ∀ i x p q t, store_event (req i x p q t) x
| put_event_spl: ∀ i xi yi p, store_event (splitp i xi yi p) xi
| put_event_spr: ∀ i xi yi p, store_event (splitp i xi yi p) yi
| get_event: ∀ i x p q, store_event (rpy i x p q) x
| get_event_joinpl: ∀ i xi yi p, store_event (joinp i xi yi p) xi
| get_event_joinpr: ∀ i xi yi p, store_event (joinp i xi yi p) yi.

Lemma store_events_lrange (t:AnnoTerm) : ∀ p ev loc,
    well_formed t →
    events t p ev →
    store_event ev loc →
    fst (lrange t) ≤ loc < snd (lrange t).

Ltac pose_store_events :=
  match goal with
  | [H: events _ _ ?ev,
        H': Loc |- _] ⇒
    assert_new_proof_by (store_event ev H') econstructor
  end.

Ltac pose_new_lrange :=
  match goal with
  | [H: well_formed ?t,
        H': events ?t ?p ?ev,
            H'': store_event ?ev ?loc
     |- _] ⇒
    pose_new_proof (store_events_lrange t p ev loc H H' H'')
  end.

Ltac pose_lrange_facts :=
  repeat pose_store_events;
  repeat pose_new_lrange;
  repeat find_eapply_lem_hyp wf_mono_locs.

Ltac dest_lrange :=
  match goal with
  | [H: LocRange |- _] ⇒ destruct H
  end.

Lemma unique_store_events: ∀ t p ev1 ev2 loc,
  well_formed t →
  events t p ev1 →
  events t p ev2 →
  store_event ev1 loc →
  store_event ev2 loc →
  ev1 = ev2.

Lemma unique_store_events': ∀ t p ev1 ev2 loc,
    well_formed t →
    ev_in ev1 (ev_sys t p) →
    ev_in ev2 (ev_sys t p) →
    store_event ev1 loc →
    store_event ev2 loc →
    ev1 ≠ ev2 →
    False.

Definition store_event_evsys es loc := ∃ ev, store_event ev loc ∧ ev_in ev es.

Inductive store_conflict: EvSys Ev → Prop :=
| store_conflict_merge: ∀ r es1 es2 loc,
    store_event_evsys es1 loc →
    store_event_evsys es2 loc →
    store_conflict (merge r es1 es2)
| store_conflict_before_l: ∀ r es1 es2,
    store_conflict es1 →
    store_conflict (before r es1 es2)
| store_conflict_before_r: ∀ r es1 es2,
    store_conflict es2 →
    store_conflict (before r es1 es2).

Lemma unique_events': ∀ r es1 es2 ev1 ev2,
    well_structured ev (merge r es1 es2) →
    ev_in ev1 es1 →
    ev_in ev2 es2 →
    ev ev1 ≠ ev ev2.

Lemma unqev: ∀ ev1 ev2,
  ev ev1 ≠ ev ev2 →
  ev1 ≠ ev2.

Lemma unique_events: ∀ r es1 es2 ev1 ev2,
    well_structured ev (merge r es1 es2) →
    ev_in ev1 es1 →
    ev_in ev2 es2 →
    ev1 ≠ ev2.

Theorem no_store_conflicts: ∀ t p sys,
    well_formed t →
    sys = ev_sys t p →
    not (store_conflict sys).

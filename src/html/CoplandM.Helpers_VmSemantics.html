Lemma pl_immut : ∀ t e tr p,
    well_formed_r t →
    st_pl
      (execSt
         (copland_compile t)
         {|
           st_ev := e;
           st_trace := tr;
           st_pl := p |}) = p.

Ltac do_pl_immut :=
  let tac H H2 :=
       erewrite <- pl_immut;
        [ unfold execSt;
          rewrite H;
          reflexivity |
          apply H2] in
      match goal with
      | [H: copland_compile ?t
            {| st_ev := _;
                        st_trace := _;
                        st_pl := ?p |} =
            (_,
             {| st_ev := _;
                         st_trace := _;
                         st_pl := ?p' |}),
         H2: well_formed_r ?t |- _] ⇒
        assert_new_proof_by (p' = p) ltac:(tac H H2)
      end.

Lemma st_congr :
  ∀ st tr e p,
    st_ev st = e →
    st_trace st = tr →
    st_pl st = p →
    
    st = {| st_ev := e; st_trace := tr; st_pl := p |}.


Ltac anhl :=
  annogo;
  match goal with
  | [H: well_formed_r ?a,
     H2: copland_compile ?a _ = _,
     H3: copland_compile ?a _ = _,
     IH: context[well_formed_r ?a → _] |- _] ⇒
    edestruct IH;
    [apply H |
     apply H2 | apply H3 | idtac]; clear H2; clear H3;
    destruct_conjs; subst
  end.

Lemma hihi : ∀ t e e' e'' x x' y y' p p' p'',
    well_formed_r t →
    copland_compile t {| st_ev := e; st_trace := x; st_pl := p |} =
    (Some tt, {| st_ev := e'; st_trace := x'; st_pl := p' |}) →
    copland_compile t {| st_ev := e; st_trace := y; st_pl := p |} =
    (Some tt, {| st_ev := e''; st_trace := y'; st_pl := p'' |}) →
    (e' = e'' ∧ p' = p'' ).

Ltac clear_triv :=
  match goal with
  | [H: ?x = ?x |- _] ⇒ clear H
  end.

Ltac dohi'' :=
  annogo;
  let tac Hwf H H' := eapply hihi; [apply Hwf | apply H | apply H'] in
  match goal with
  | [H : copland_compile ?t1 {| st_ev := ?e; st_trace := _; st_pl := ?p |} =
         (?opt, {| st_ev := ?e'; st_trace := _; st_pl := ?p' |}),
         H' : copland_compile ?t1 {| st_ev := ?e; st_trace := _; st_pl := ?p |} =
              (?opt, {| st_ev := ?e''; st_trace := _; st_pl := ?p'' |}),
                Hwf : well_formed_r ?t1 |- _] ⇒
    assert_new_proof_by (e' = e'' ∧ p' = p'' ) ltac:(tac Hwf H H')
  end.

Ltac dohi :=
  do 2 (repeat dohi''; destruct_conjs; subst);
  clear_triv.

Require Import List_Facts.


Lemma always_some : ∀ t vm_st vm_st' op,
    well_formed_r t →
    copland_compile
      t
      vm_st =
    (op, vm_st') →
    op = Some tt.

Ltac do_somett :=
  match goal with
  | [H: copland_compile ?t _ = (?o, _),
        H2: well_formed_r ?t |- _] ⇒
    assert_new_proof_by (o = Some tt) ltac:(eapply always_some; [apply H2 | apply H])
  end.

Ltac do_asome := repeat do_somett; clear_triv.


Lemma trace_irrel_pl : ∀ t tr1 tr1' tr2 e e' p1' p1,
    well_formed_r t →
    copland_compile t
           {| st_ev := e; st_trace := tr1; st_pl := p1 |} =
    (Some tt, {| st_ev := e'; st_trace := tr1'; st_pl := p1' |}) →
    
    st_pl
      (execSt (copland_compile t)
           {| st_ev := e; st_trace := tr2; st_pl := p1 |}) = p1'.

Require Import Coq.Arith.Even.

Lemma both_args_even: ∀ a b,
    Nat.even a = true →
    Nat.even b = true →
    Nat.even (a + b) = true.

Definition get_available_locs (n:nat): list nat.
Admitted.

Axiom num_locs: ∀ n l,
    get_available_locs n = l →
    length l = n.

Axiom unique_locs: ∀ n l,
  get_available_locs n = l →
  NoDup l.

Fixpoint make_pairs' (l:list nat) (pl:list (nat×nat)): option (list (nat×nat)) :=
  match l with
  | [] ⇒ Some pl
  | x :: y :: l' ⇒ make_pairs' l' (pl ++ [(x,y)])
  | _ ⇒ None
  end.

Definition make_pairs (l:list nat): option (list (nat×nat)) :=
  make_pairs' l [].


Lemma even_list_struc{A:Type}: ∀ n (ls:list A),
    length ls = n →
    Coq.Arith.Even.even n →
    ls = [] ∨ ∃ x y ls', ls = x :: y :: ls' ∧ (Coq.Arith.Even.even (length ls')).


Definition even_list_ind{A:Type}
  : ∀ P : list A → Prop, P [] →
                    (∀ ls x y, even (length ls) → P ls → P (x :: y :: ls)) →
                    (∀ ls, even (length ls) → P ls).
Admitted.



Lemma cumul: ∀ ls ps H0,
    Coq.Arith.Even.even (length ls) →
    make_pairs' ls ps = Some H0 →
    ∃ lss,
    H0 = ps ++ lss.

Lemma make_pairs_succeeds: ∀ ps ls,
  
  Coq.Arith.Even.even (length ls) →
  ∃ ps',
  make_pairs' ls ps = Some ps'.

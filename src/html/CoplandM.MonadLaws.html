Require Import GenStMonad.

Require Import StructTactics.

Lemma monad_left_id : ∀ S A B (a:A)(f:A → (GenStMonad.St S) B) (s:S),
    (bind (ret a) f) s = (f a s).

Lemma monad_right_id : ∀ S A (m:St S A) (s:S),
    (bind m ret) s = m s.

Lemma monad_right_id'{S A:Type} : ∀ (m:St S A) (s:S),
    (v <- m ;; (ret v)) s = m s.

Lemma monad_right_id''{S:Type} : ∀ (m:St S unit) (s:S),
    (m ;; (ret tt)) s = m s.

Lemma monad_comp{S A B C:Type} : ∀ (m: St S A) (k:A → St S B) (h:B → St S C) (s:S),
    bind m (fun x ⇒ (bind (k x) h)) s = (bind (bind m k) h) s.

Lemma monad_get_get{S A:Type} : ∀ (k: St S A) (st:S),
    (get ;; get ;; k) st = (get ;; k) st.

Lemma monad_get_put{A:Type} : ∀ (st: A),
    (s <- get ;; put s) st = (ret tt) st.

Lemma monad_put_put{S:Type} : ∀ (st s s':S),
    (put s ;; put s') st = (put s') st.

Lemma monad_put_get{S A:Type} : ∀ (st s: S) (k:St S A),
    (put s;; get ;; k) st = (put s ;; k) st.

Lemma gasd{A B:Type} : ∀ (act:St A B) (act2:St A B) st,
    (act ;; ret tt ;; act2) st =
    (act ;; act2) st.

Lemma fafa{A B:Type} : ∀ (act act2 act3: St A B) st,
    ((act;; ret tt;; act2);;
     act3) st =
    ((act;; act2);;
     act3) st.

Lemma hlhl{A B:Type} : ∀ (act act2 act3 act4 : St A B) st,
    ((act;; act2;; act3);;
     act4) st =
    (((act;; act2);; act3);;
     act4) st.

Lemma hghg{A B:Type} : ∀ (act act2 act3 act4 act5 : St A B) st,
    (((act;; act2;; act3);; act5);;
     act4) st =
    ((act;; act2;; act3);; act5;; act4) st.

Library CoplandM.MonadVMFacts


Require Import Term ConcreteEvidence MonadVM Maps.
Require Import Coq.Program.Tactics.

Require Import List.
Import ListNotations.

Require Import StructTactics.

Ltac vmsts :=
  simpl in *;
  repeat
    match goal with
    | [H: cvm_st |- _] ⇒ destruct H
    end.

Ltac amsts :=
  repeat match goal with
         | H:cvm_st |- _ ⇒ destruct H
         end.

Ltac pairs :=
  simpl in *;
  vmsts;
  repeat
    match goal with
    | [H: (Some _, _) =
          (Some _, _) |- _ ] ⇒ invc H; monad_unfold
                                                          
    | [H: {| st_ev := _; st_trace := _; st_pl := _ |} =
          {| st_ev := _; st_trace := _; st_pl := _ |} |- _ ] ⇒
      invc H; monad_unfold
    end; destruct_conjs; monad_unfold.

Construct an event system from an annotated term, place, and evidence.

Fixpoint ev_sys (t: AnnoTerm) p: EvSys Ev :=
  match t with
  | aasp (i, j) lr x ⇒ leaf (i, j) (asp_event i x p)
  | aatt (i, j) lr (req_loc,rpy_loc) q x ⇒
    before (i, j)
      (leaf (i, S i) (req i req_loc p q (unanno x)))
      (before (S i, j)
              (ev_sys x q)
              (leaf (pred j, j) (rpy (pred j) rpy_loc p q)))
  | alseq r lr x y ⇒ before r (ev_sys x p)
                          (ev_sys y p)
  
  end.

Lemma evsys_range:
  ∀ t p,
    es_range (ev_sys t p) = range t.

Lemma well_structured_evsys:
  ∀ t p,
    well_formed_r t →
    well_structured ev (ev_sys t p).

The events in the event system correspond to the events associated with a term, a place, and some evidence.

Lemma evsys_events:
  ∀ t p ev,
    well_formed_r t →
    ev_in ev (ev_sys t p) ↔ events t p ev.

  match goal with

  end.

Maximal events are unique.

Lemma supreme_unique:
  ∀ t p,
    well_formed_r t →
    ∃ ! v, supreme (ev_sys t p) v.

Lemma evsys_max_unique:
  ∀ t p,
    well_formed_r t →
    unique (supreme (ev_sys t p)) (max (ev_sys t p)).

Lemma firstn_gt: ∀ (ls: list nat) m n,
    m ≥ n →
    firstn n (firstn m ls) = firstn n ls.

Lemma skipn_firstn_len: ∀ (ls:list nat) n m,

    (skipn m (firstn (m + n) ls)).

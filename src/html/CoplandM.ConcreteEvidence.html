Notation BS := nat (only parsing).

Concrete Evidence

Inductive EvidenceC: Set :=
| mtc: EvidenceC
| uuc: ASP_ID → list Arg → Plc → BS → EvidenceC → EvidenceC
| ggc: Plc → BS → EvidenceC → EvidenceC

| nnc: N_ID → BS → EvidenceC → EvidenceC

.

Inductive EvSub: EvidenceC → EvidenceC → Prop :=
| evsub_refl : ∀ e : EvidenceC, EvSub e e
| uuSub: ∀ e e' i l p bs,
    EvSub e e' →
    EvSub e (uuc i l p bs e')
| ggSub: ∀ e e' p bs,
    EvSub e e' →
    EvSub e (ggc p bs e')
| nnSub: ∀ e e' i bs,
    EvSub e e' →
    EvSub e (nnc i bs e').


Fixpoint et_fun (ec:EvidenceC) : Evidence :=
  match ec with
  | mtc ⇒ mt
  | uuc i l p _ ec' ⇒ uu i l p (et_fun ec')
  | ggc p _ ec' ⇒ gg p (et_fun ec')
  
  | nnc ni _ ec' ⇒ nn ni (et_fun ec')
  
  end.

Inductive Ev_Shape: EvidenceC → Evidence → Prop :=
| mtt: Ev_Shape mtc mt
| uut: ∀ id args p bs e et,
    Ev_Shape e et →
    Ev_Shape (uuc id args p bs e) (uu id args p et)
| ggt: ∀ p bs e et,
    Ev_Shape e et →
    Ev_Shape (ggc p bs e) (gg p et)

| nnt: ∀ bs e et i,
    Ev_Shape e et →
    Ev_Shape (nnc i bs e) (nn i et)

.
Hint Constructors Ev_Shape : core.

Ltac evShapeFacts :=
  match goal with
  | [H: Ev_Shape mtc _ |- _] ⇒ invc H
  | [H: Ev_Shape _ mt |- _] ⇒ invc H
  | [H: Ev_Shape (uuc _ _ _ _ _) _ |- _] ⇒ invc H
  | [H: Ev_Shape _ (uu _ _ _ _) |- _] ⇒ invc H
  | [H: Ev_Shape (ggc _ _ _) _ |- _] ⇒ invc H
  | [H: Ev_Shape _ (gg _ _) |- _] ⇒ invc H
  
  | [H: Ev_Shape (nnc _ _ _) _ |- _] ⇒ invc H
  | [H: Ev_Shape _ (nn _ _) |- _] ⇒ invc H
  
  
  end.

Lemma ev_evshape: ∀ ec,
    Ev_Shape ec (et_fun ec).

Lemma evshape_determ: ∀ ec et et',
  Ev_Shape ec et →
  Ev_Shape ec et' →
  et = et'.







Lemma ev_shape_transitive : ∀ e e' et et',
    Ev_Shape e et →
    Ev_Shape e' et →
    Ev_Shape e et' →
    Ev_Shape e' et'.


Eval function definition

Definition splitEv (sp:SP) (e:EvidenceC) : EvidenceC :=
  match sp with
  | ALL ⇒ e
  | NONE ⇒ mtc
  end.

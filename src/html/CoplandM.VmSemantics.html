Lemma store_events_injective: ∀ t p ev1 ev2 loc,
  well_formed t →
  events t p ev1 →
  events t p ev2 →
  store_event ev1 loc →
  store_event ev2 loc →
  ev1 = ev2.



Lemma unique_store_events': ∀ t p ev1 ev2 loc,
    well_formed t →
    ev_in ev1 (Term_system.ev_sys t p) →
    ev_in ev2 (Term_system.ev_sys t p) →
    store_event ev1 loc →
    store_event ev2 loc →
    ev1 ≠ ev2 →
    False.

Inductive store_event_evsys: EvSys Ev → Loc → Prop :=
| inevsys_leaf: ∀ r ev loc,
    store_event ev loc →
    store_event_evsys (leaf r ev) loc
| inevsys_before_l: ∀ r es1 es2 loc,
    store_event_evsys es1 loc →
    store_event_evsys (before r es1 es2) loc
| inevsys_before_r: ∀ r es1 es2 loc,
    store_event_evsys es2 loc →
    store_event_evsys (before r es1 es2) loc
| inevsys_merge_l: ∀ r es1 es2 loc,
    store_event_evsys es1 loc →
    store_event_evsys (merge r es1 es2) loc
| inevsys_merge_r: ∀ r es1 es2 loc,
    store_event_evsys es2 loc →
    store_event_evsys (merge r es1 es2) loc.

Inductive store_conflict: EvSys Ev → Prop :=
| store_conflict_merge: ∀ r es1 es2 loc,
    store_event_evsys es1 loc →
    store_event_evsys es2 loc →
    store_conflict (merge r es1 es2)
| store_conflict_before_l: ∀ r es1 es2,
    store_conflict es1 →
    store_conflict (before r es1 es2)
| store_conflict_before_r: ∀ r es1 es2,
    store_conflict es2 →
    store_conflict (before r es1 es2).

Lemma unique_events': ∀ r es1 es2 ev1 ev2,

Lemma suffix_prop : ∀ t e e' tr tr' p p' o o',
    well_formed t →
    copland_compile t
           {|
             st_ev := e;
             st_trace := tr;
             st_pl := p;
             st_store := o |} =
    (Some tt, {|
      st_ev := e';
      st_trace := tr';
      st_pl := p';
      st_store := o' |}) →
    ∃ l, tr' = tr ++ l.

Ltac do_suffix name :=

  | [H': copland_compile ?t
         {| st_ev := _;st_trace := ?tr; st_pl := _; st_store := _ |} =
         (Some tt,
          {| st_ev := _; st_trace := ?tr'; st_pl := _; st_store := _ |}),
         H2: well_formed ?t |- _] ⇒

  end.

Lemma alseq_decomp : ∀ r lr t1' t2' e e'' p p'' o o'' tr,
    well_formed (alseq r lr t1' t2') →
    copland_compile (alseq r lr t1' t2') {| st_ev := e; st_trace := []; st_pl := p; st_store := o |} =
    (Some tt, {| st_ev := e''; st_trace := tr; st_pl := p''; st_store := o'' |}) →

    ∃ e' tr' p' o',
      copland_compile t1' {| st_ev := e; st_trace := []; st_pl := p; st_store := o |} =
      (Some tt, {| st_ev := e'; st_trace := tr'; st_pl := p'; st_store := o' |}) ∧
      ∃ tr'',
        copland_compile t2' {| st_ev := e'; st_trace := []; st_pl := p'; st_store := o' |} =
        (Some tt, {| st_ev := e''; st_trace := tr''; st_pl := p''; st_store := o'' |}) ∧
        tr = tr' ++ tr''.

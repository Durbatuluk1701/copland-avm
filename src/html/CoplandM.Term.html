Lemma wf_lseq_pieces: ∀ r t1 t2,
    well_formed_r (alseq r t1 t2) →
    well_formed_r t1 ∧ well_formed_r t2.

Lemma wf_at_pieces: ∀ t r p,
    well_formed_r (aatt r p t) →
    well_formed_r t.


Ltac do_wf_pieces :=
  match goal with
  | [H: well_formed_r (alseq _ _ _ ) |- _] ⇒
    (edestruct wf_lseq_pieces; eauto)
  | [H: well_formed_r (aatt _ _?t) |- _] ⇒
    assert (well_formed_r t)
      by (eapply wf_at_pieces; eauto)
  
  end.

Eval for annotated terms.

Fixpoint aeval t p e :=
  match t with
  | aasp _ x ⇒ eval (asp x) p e
  | aatt _ q x ⇒ aeval x q e
  | alseq _ t1 t2 ⇒ aeval t2 p (aeval t1 p e)
  
  end.


This predicate specifies when a term, a place, and some initial evidence is related to an event. In other words, it specifies the set of events associated with a term, a place, and some initial evidence.

Inductive events: AnnoTerm → Plc → Ev → Prop :=
| evtscpy:
    ∀ r i p,
      fst r = i →
      events (aasp r CPY) p (copy i p)
| evtsusm:
    ∀ i id args r p,
      fst r = i →
      events (aasp r (ASPC id args)) p (umeas i p id args)
| evtssig:
    ∀ r i p,
      fst r = i →
      events (aasp r SIG) p (sign i p)

| evtsattreq:
    ∀ r q t i p,
      fst r = i →
      events (aatt r q t) p (req i p q (unanno t))
| evtsatt:
    ∀ r q t ev p,
      events t q ev →
      events (aatt r q t) p ev
| evtsattrpy:
    ∀ r q t i p,
      snd r = S i →
      events (aatt r q t) p (rpy i p q)
| evtslseql:
    ∀ r t1 t2 ev p,
      events t1 p ev →
      events (alseq r t1 t2) p ev
| evtslseqr:
    ∀ r t1 t2 ev p,
      events t2 p ev →
      events (alseq r t1 t2) p ev
              .
Hint Constructors events : core.

Lemma events_range:
  ∀ t v p,
    well_formed_r t →
    events t p v →
    fst (range t) ≤ ev v < snd (range t).

Lemma at_range:
  ∀ x r i,
    S (fst r) = fst x →
    snd r = S (snd x) →
    fst r ≤ i < snd r →
    i = fst r ∨
    fst x ≤ i < snd x ∨
    i = snd x.

Lemma lin_range:
  ∀ x y i,
    snd x = fst y →
    fst x ≤ i < snd y →
    fst x ≤ i < snd x ∨
    fst y ≤ i < snd y.

Lemma bra_range:
  ∀ x y r i,
    S (fst r) = fst x →
    snd x = fst y →
    snd r = S (snd y) →
    fst r ≤ i < snd r →
    i = fst r ∨
    fst x ≤ i < snd x ∨
    fst y ≤ i < snd y ∨
    i = snd y.

Lemma events_range_event:
  ∀ t p i,
    well_formed_r t →
    fst (range t) ≤ i < snd (range t) →
    ∃ v, events t p v ∧ ev v = i.
    Ltac dest_range :=
      match goal with
      | [H: (nat × nat) |- _] ⇒ destruct H
      end.
     Ltac do_lin_range :=
      match goal with
      | [H: snd _ = fst _,
            H': fst _ ≤ ?n < snd _
         |- _] ⇒
        apply lin_range with (i:=n) in H; eauto
      end.
     Ltac do_bra_range :=
      match goal with
      | [H: snd _ = fst _,
            H': fst ?x ≤ ?n < snd ?x
         |- _] ⇒
        apply bra_range with (i:=n) (r:=x) in H; eauto
      end.

Ltac events_event_range :=
  repeat match goal with
         | [ H: events _ _ _ |- _ ] ⇒
           apply events_range in H; auto
         end; lia.

Ltac aba :=
  match goal with
  | [H: events _ _ _, H': events _ _ _ |- _] ⇒ inv H; inv H'
  end.

Ltac wfr :=
  match goal with
  | [ H': well_formed_r ?HH |- _] ⇒ pose_new_proof (well_formed_range_r HH H')
  end.

Lemma events_injective:
  ∀ t p v1 v2,
    well_formed_r t →
    events t p v1 →
    events t p v2 →
    ev v1 = ev v2 →
    v1 = v2.

Inductive evalR : Term → Plc → Evidence → Evidence → Prop :=
| evalR_asp: ∀ a p e,
    evalR (asp a) p e (eval_asp a p e)
| evalR_att: ∀ t1 q e e' p,
    evalR t1 q e e' →
    evalR (att q t1) p e e'
| evalR_lseq: ∀ t1 t2 p e e' e'',
    evalR t1 p e e' →
    evalR t2 p e' e'' →
    evalR (lseq t1 t2) p e e''
